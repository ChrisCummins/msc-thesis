\section{Algebra}

\subsection{Laws of indices}

\begin{align}
a^m \times a^n &= a^{m+n}\\
\frac{a^m}{a^n} &= a^{m-n}\\
\left(a^m\right)^n &= a^{mn}
\end{align}

\subsection{Distributive laws}

\setcounter{equation}{0}
\begin{align}
a(b + c) &= ab + ac\\
a(b - c) &= ab - ac
\end{align}

\subsection{Bracket Expansion}
\begin{align}
(a+b)(c+d) &= c(a+b) + d(a+b)\\
&= ca + cb + da + db
\end{align}
